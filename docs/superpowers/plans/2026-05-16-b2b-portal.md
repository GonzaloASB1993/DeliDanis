# B2B Portal for Cafeterias — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a wholesale ordering portal at `/b2b` where cafeterias (accounts created by admin) browse the catalog with B2B pricing, build orders by quantity, view order history, and repeat past orders.

**Architecture:** New route group `app/b2b/` with its own layout (minimal navbar, no sidebar). Reuses existing product tables (`cake_products`, `pastry_products`, `cocktail_products`) with a new `b2b_prices` table for wholesale pricing. Orders go into the existing `orders` table with a `channel` column to distinguish B2B from public. A new `b2b_client` role in `user_profiles` gates access via middleware.

**Tech Stack:** Next.js 14 App Router, Supabase (auth + DB), Zustand (cart store), Tailwind CSS, existing `supabaseAdmin` for user creation API.

```sql
-- 1. B2B prices table
CREATE TABLE IF NOT EXISTS b2b_prices (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  product_id UUID NOT NULL,
  product_type VARCHAR(20) NOT NULL, -- 'cake', 'pastry', 'cocktail'
  price DECIMAL(10,2) NOT NULL,
  min_quantity INTEGER DEFAULT 1,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(product_id, product_type)
);

CREATE INDEX idx_b2b_prices_product ON b2b_prices(product_id, product_type);
CREATE INDEX idx_b2b_prices_active ON b2b_prices(is_active);

-- 2. Add customer type column
ALTER TABLE customers ADD COLUMN IF NOT EXISTS type VARCHAR(20) DEFAULT 'particular';
-- values: 'particular', 'business'

-- 3. Add order channel column
ALTER TABLE orders ADD COLUMN IF NOT EXISTS channel VARCHAR(20) DEFAULT 'public';
-- values: 'public', 'b2b'

-- 4. RLS for b2b_prices
ALTER TABLE b2b_prices ENABLE ROW LEVEL SECURITY;

-- Admin/manager can do everything with b2b_prices
CREATE POLICY "Admin manages b2b_prices" ON b2b_prices
  FOR ALL USING (
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role IN ('admin', 'manager') AND is_active = true
    )
  );

-- B2B clients can read active prices
CREATE POLICY "B2B clients read active prices" ON b2b_prices
  FOR SELECT USING (
    is_active = true AND
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role = 'b2b_client' AND is_active = true
    )
  );

-- 5. B2B clients can read products
CREATE POLICY "B2B clients read cake_products" ON cake_products
  FOR SELECT USING (
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role = 'b2b_client' AND is_active = true
    )
  );

CREATE POLICY "B2B clients read pastry_products" ON pastry_products
  FOR SELECT USING (
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role = 'b2b_client' AND is_active = true
    )
  );

CREATE POLICY "B2B clients read cocktail_products" ON cocktail_products
  FOR SELECT USING (
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role = 'b2b_client' AND is_active = true
    )
  );

-- 6. B2B clients can read product_images
CREATE POLICY "B2B clients read product_images" ON product_images
  FOR SELECT USING (
    auth.uid() IN (
      SELECT id FROM user_profiles WHERE role = 'b2b_client' AND is_active = true
    )
  );

-- 7. B2B clients can insert orders (their own)
CREATE POLICY "B2B clients create orders" ON orders
  FOR INSERT WITH CHECK (
    auth.uid() IN (
      SELECT user_id FROM customers WHERE id = customer_id AND type = 'business'
    )
  );

-- 8. B2B clients can read their own orders
CREATE POLICY "B2B clients read own orders" ON orders
  FOR SELECT USING (
    channel = 'b2b' AND
    customer_id IN (
      SELECT id FROM customers WHERE user_id = auth.uid()
    )
  );

-- 9. B2B clients can insert order_items for their orders
CREATE POLICY "B2B clients create order_items" ON order_items
  FOR INSERT WITH CHECK (
    order_id IN (
      SELECT id FROM orders WHERE customer_id IN (
        SELECT id FROM customers WHERE user_id = auth.uid()
      )
    )
  );

-- 10. B2B clients can read their order_items
CREATE POLICY "B2B clients read own order_items" ON order_items
  FOR SELECT USING (
    order_id IN (
      SELECT id FROM orders WHERE customer_id IN (
        SELECT id FROM customers WHERE user_id = auth.uid()
      )
    )
  );
```

```typescript
  const isB2BRoute = request.nextUrl.pathname.startsWith('/b2b')

  // Unauthenticated users trying to access B2B portal
  if (isB2BRoute && !isB2BLogin && !user) {
    const url = request.nextUrl.clone()

      .from('user_profiles')
      .select('role')
      .eq('id', user.id)
      .single()

    if (profile?.role === 'b2b_client') {
      const url = request.nextUrl.clone()
      url.pathname = '/b2b'
      return NextResponse.redirect(url)
    }
  }

  // Authenticated non-B2B user trying to access B2B portal
  if (isB2BRoute && !isB2BLogin && user) {
    const { data: profile } = await supabase
      .from('user_profiles')
      .select('role, is_active')
      .eq('id', user.id)
      .single()

    if (!profile || profile.role !== 'b2b_client' || !profile.is_active) {
      const url = request.nextUrl.clone()
      url.pathname = '/b2b/login'
      url.searchParams.set('error', 'unauthorized')
      return NextResponse.redirect(url)
    }
  }
```

```typescript
import { supabase } from './client'
import type { B2BProduct, B2BOrderSummary, B2BOrderDetail } from '@/types/b2b'

/**
 * Fetch all products that have active B2B prices.
 * Combines cake, pastry, cocktail products with their b2b_prices.
 */
export async function getB2BProducts(): Promise<{ success: boolean; products: B2BProduct[] }> {
  try {
    const { data: prices, error: pricesError } = await supabase
      .from('b2b_prices')
      .select('*')
      .eq('is_active', true)

    if (pricesError || !prices?.length) {
      return { success: !pricesError, products: [] }
    }

    const products: B2BProduct[] = []

    // Group prices by product_type
    const cakeIds = prices.filter(p => p.product_type === 'cake').map(p => p.product_id)
    const pastryIds = prices.filter(p => p.product_type === 'pastry').map(p => p.product_id)
    const cocktailIds = prices.filter(p => p.product_type === 'cocktail').map(p => p.product_id)

    // Fetch cake products
    if (cakeIds.length) {
      const { data } = await supabase
        .from('cake_products')
        .select('id, name, slug, description, short_description, image_url, category:cake_categories(name)')
        .in('id', cakeIds)
        .eq('is_active', true)

      data?.forEach(p => {
        const price = prices.find(pr => pr.product_id === p.id && pr.product_type === 'cake')!
        products.push({
          id: p.id,
          name: p.name,
          slug: p.slug,
          description: p.description,
          short_description: p.short_description,
          product_type: 'cake',
          category_name: (p.category as any)?.name ?? null,
          image_url: p.image_url,
          b2b_price: price.price,
          min_quantity: price.min_quantity,
          is_active: true,
        })
      })
    }

    // Fetch pastry products
    if (pastryIds.length) {
      const { data } = await supabase
        .from('pastry_products')
        .select('id, name, slug, description, short_description, image_url, category:pastry_categories(name)')
        .in('id', pastryIds)
        .eq('is_active', true)

      data?.forEach(p => {
        const price = prices.find(pr => pr.product_id === p.id && pr.product_type === 'pastry')!
        products.push({
          id: p.id,
          name: p.name,
          slug: p.slug,
          description: p.description,
          short_description: p.short_description,
          product_type: 'pastry',
          category_name: (p.category as any)?.name ?? null,
          image_url: p.image_url,
          b2b_price: price.price,
          min_quantity: price.min_quantity,
          is_active: true,
        })
      })
    }

    // Fetch cocktail products
    if (cocktailIds.length) {
      const { data } = await supabase
        .from('cocktail_products')
        .select('id, name, slug, description, short_description, image_url, category:cocktail_categories(name)')
        .in('id', cocktailIds)
        .eq('is_active', true)

      data?.forEach(p => {
        const price = prices.find(pr => pr.product_id === p.id && pr.product_type === 'cocktail')!
        products.push({
          id: p.id,
          name: p.name,
          slug: p.slug,

          b2b_price: price.price,
          min_quantity: price.min_quantity,
          is_active: true,
        })
      })
    }

    return { success: true, products }
  } catch (error) {
    console.error('Error loading B2B products:', error)
    return { success: false, products: [] }
  }
}

/**
 * Create a B2B order from cart items.
 */
export async function createB2BOrder(
  customerId: string,
  items: { productId: string; productName: string; quantity: number; unitPrice: number }[]
): Promise<{ success: boolean; orderId?: string; orderNumber?: string; error?: string }> {
  try {
    const subtotal = items.reduce((sum, item) => sum + item.quantity * item.unitPrice, 0)
    const orderNumber = `B2B-${Date.now().toString(36).toUpperCase()}`

    const { data: order, error: orderError } = await supabase
      .from('orders')
      .insert({
        order_number: orderNumber,
        customer_id: customerId,
        status: 'pending',
        channel: 'b2b',
        event_type: 'b2b_order',
        event_date: new Date().toISOString().split('T')[0],
        delivery_type: 'pickup',
        subtotal,
        discount: 0,
        total: subtotal,
        payment_status: 'pending',
      })
      .select('id, order_number')
      .single()

    if (orderError) {
      console.error('Error creating B2B order:', orderError)
      return { success: false, error: orderError.message }
    }

    const orderItems = items.map(item => ({
      order_id: order.id,
      product_id: item.productId,
      product_name: item.productName,
      quantity: item.quantity,
      unit_price: item.unitPrice,
      total_price: item.quantity * item.unitPrice,
    }))

    const { error: itemsError } = await supabase
      .from('order_items')
      .insert(orderItems)

    if (itemsError) {
      console.error('Error creating order items:', itemsError)
      return { success: false, error: itemsError.message }
    }

    return { success: true, orderId: order.id, orderNumber: order.order_number }
  } catch (error) {
    console.error('Error in createB2BOrder:', error)
    return { success: false, error: 'Error inesperado' }
  }
}

/**
 * Get order history for current B2B user.
 */
export async function getB2BOrders(): Promise<{ success: boolean; orders: B2BOrderSummary[] }> {
  try {
    const customer = await getB2BCustomer()
    if (!customer) return { success: false, orders: [] }

    const { data, error } = await supabase
      .from('orders')
      .select('id, order_number, status, total, created_at, order_items(id)')
      .eq('customer_id', customer.id)
      .eq('channel', 'b2b')
      .order('created_at', { ascending: false })

    if (error) {
      console.error('Error fetching B2B orders:', error)
      return { success: false, orders: [] }
    }

    const orders: B2BOrderSummary[] = (data || []).map(o => ({
      id: o.id,
      order_number: o.order_number,
      status: o.status,
      total: o.total,
      item_count: (o.order_items as any[])?.length ?? 0,
      created_at: o.created_at,
    }))

    return { success: true, orders }
  } catch (error) {
    console.error('Error in getB2BOrders:', error)
    return { success: false, orders: [] }
  }
}

  productType: string,
  price: number,
  minQuantity: number,
  isActive: boolean
) {
  const { data, error } = await supabase
    .from('b2b_prices')

        product_type: productType,
        price,

    .single()

    return null
  }
  return data
}
```

```typescript
// stores/b2bCartStore.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import type { B2BCartItem } from '@/types/b2b'

interface B2BCartStore {
  items: B2BCartItem[]
  addItem: (item: B2BCartItem) => void
  updateQuantity: (productId: string, quantity: number) => void
  removeItem: (productId: string) => void
  clear: () => void
  getTotal: () => number
  getItemCount: () => number
}

export const useB2BCartStore = create<B2BCartStore>()(
  persist(
    (set, get) => ({
      items: [],

      addItem: (item) => {
        set((state) => {
          const existing = state.items.find(i => i.productId === item.productId)
          if (existing) {
            return {
              items: state.items.map(i =>
                i.productId === item.productId
                  ? { ...i, quantity: i.quantity + item.quantity }
                  : i
              ),
            }
          }
          return { items: [...state.items, item] }
        })
      },

      updateQuantity: (productId, quantity) => {
        set((state) => ({
          items: state.items.map(i =>
            i.productId === productId ? { ...i, quantity } : i
          ),
        }))
      },

      removeItem: (productId) => {
        set((state) => ({
          items: state.items.filter(i => i.productId !== productId),
        }))
      },

      clear: () => set({ items: [] }),

      getTotal: () => {
        return get().items.reduce((sum, i) => sum + i.quantity * i.unitPrice, 0)
      },

      getItemCount: () => {
        return get().items.reduce((sum, i) => sum + i.quantity, 0)
      },
    }),
    {
      name: 'b2b-cart',
    }
  )
)
```

```typescript
'use client'

import { useState } from 'react'
import { useRouter, useSearchParams } from 'next/navigation'
import { supabase } from '@/lib/supabase/client'

export function B2BLoginForm() {
  const router = useRouter()
  const searchParams = useSearchParams()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState('')
  const [loading, setLoading] = useState(false)

  const urlError = searchParams.get('error')

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setError('')
    setLoading(true)

    const { data, error: authError } = await supabase.auth.signInWithPassword({
      email,
      password,
    })

    if (authError) {
      setError('Email o contraseña incorrectos')
      setLoading(false)
      return
    }

    // Verify role is b2b_client
    const { data: profile } = await supabase
      .from('user_profiles')
      .select('role, is_active')
      .eq('id', data.user.id)
      .single()

    if (!profile || profile.role !== 'b2b_client') {
      await supabase.auth.signOut()
      setError('Esta cuenta no tiene acceso al portal B2B')
      setLoading(false)
      return
    }

    if (!profile.is_active) {
      await supabase.auth.signOut()
      setError('Tu cuenta está desactivada. Contacta al administrador.')
      setLoading(false)
      return
    }

    router.push('/b2b')
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-5">
      {(error || urlError === 'unauthorized') && (
        <div className="p-3 bg-red-50 border border-red-200 rounded-lg text-sm text-red-700">
```

```typescript
import { useState, useMemo } from 'react'
import { B2BProductCard } from './B2BProductCard'
import type { B2BProduct } from '@/types/b2b'

interface B2BProductGridProps {
  products: B2BProduct[]
}

export function B2BProductGrid({ products }: B2BProductGridProps) {
  const [search, setSearch] = useState('')
  const [typeFilter, setTypeFilter] = useState<string>('all')

  const types = useMemo(() => {
    const set = new Set(products.map((p) => p.product_type))
    return Array.from(set)
  }, [products])

  const typeLabels: Record<string, string> = {
    cake: 'Tortas',
    pastry: 'Pastelería',
    cocktail: 'Coctelería',
  }

  const filtered = useMemo(() => {
    return products.filter((p) => {
      const matchesSearch = !search || p.name.toLowerCase().includes(search.toLowerCase())
      const matchesType = typeFilter === 'all' || p.product_type === typeFilter
      return matchesSearch && matchesType
    })
  }, [products, search, typeFilter])

  return (
    <div>
      {/* Filters */}
      <div className="flex flex-col sm:flex-row gap-3 mb-6">
        <input
          type="text"
          placeholder="Buscar producto..."
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          className="flex-1 px-4 py-2.5 border border-border rounded-lg focus:ring-2 focus:ring-primary/30 focus:border-primary outline-none text-sm"
        />
        <div className="flex gap-2">
          <button
            onClick={() => setTypeFilter('all')}
            className={`px-4 py-2 text-sm rounded-lg border transition-colors ${
              typeFilter === 'all'
                ? 'bg-primary text-white border-primary'
                : 'border-border text-dark-light hover:border-primary'
            }`}
          >
            Todos
          </button>

              }`}

              {typeLabels[type] || type}

      {filtered.length === 0 ? (

          No se encontraron productos.

        <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-4 gap-4">
```

```typescript
import { getOrderItemsForRepeat } from '@/lib/supabase/b2b-queries'
import { toast } from '@/stores/toastStore'
import { useRouter } from 'next/navigation'
import type { B2BOrderSummary } from '@/types/b2b'

const STATUS_LABELS: Record<string, { label: string; color: string }> = {
  pending: { label: 'Pendiente', color: 'bg-yellow-100 text-yellow-800' },
  confirmed: { label: 'Confirmado', color: 'bg-blue-100 text-blue-800' },
  in_production: { label: 'En producción', color: 'bg-purple-100 text-purple-800' },
  ready: { label: 'Listo', color: 'bg-green-100 text-green-800' },
  delivered: { label: 'Entregado', color: 'bg-green-100 text-green-800' },
  completed: { label: 'Completado', color: 'bg-gray-100 text-gray-800' },
  cancelled: { label: 'Cancelado', color: 'bg-red-100 text-red-800' },
}

interface B2BOrderListProps {
  orders: B2BOrderSummary[]
}

export function B2BOrderList({ orders }: B2BOrderListProps) {
  const router = useRouter()
  const addItem = useB2BCartStore((s) => s.addItem)
  const clear = useB2BCartStore((s) => s.clear)

  const handleRepeat = async (orderId: string) => {
    const items = await getOrderItemsForRepeat(orderId)
    if (!items.length) {
      toast.error('No se pudieron cargar los items del pedido')
      return
    }
    clear()
    items.forEach((item) => {
      addItem({
        productId: item.product_id,
        productType: 'cake',
        productName: item.product_name,
        imageUrl: null,
        quantity: item.quantity,
        unitPrice: item.unit_price,
        minQuantity: 1,
      })
    })
    toast.success('Items cargados en el carrito')
    router.push('/b2b/carrito')
  }

  if (orders.length === 0) {
    return (
      <div className="text-center py-12 text-dark-light">
        <p>Todavía no tenés pedidos.</p>
        <Link href="/b2b" className="inline-block mt-4 text-primary hover:underline">
          Ir al catálogo
        </Link>
      </div>
    )
  }

  return (
    <div className="space-y-3">
      {orders.map((order) => {
        const status = STATUS_LABELS[order.status] || { label: order.status, color: 'bg-gray-100 text-gray-800' }
        return (
          <div key={order.id} className="bg-white rounded-xl border border-border p-5 flex flex-col sm:flex-row sm:items-center justify-between gap-4">
            <div className="flex-1">
              <div className="flex items-center gap-3">
                <Link href={`/b2b/pedidos/${order.id}`} className="font-medium text-dark hover:text-primary transition-colors">
                  #{order.order_number}
                </Link>
                <span className={`text-xs px-2.5 py-0.5 rounded-full font-medium ${status.color}`}>
                  {status.label}
                </span>
              </div>
              <div className="flex gap-4 mt-1.5 text-sm text-dark-light">
                <span>{new Date(order.created_at).toLocaleDateString('es-AR')}</span>
                <span>{order.item_count} productos</span>
                <span className="font-medium text-dark">${order.total.toLocaleString('es-AR')}</span>
              </div>
            </div>
            <button
              onClick={() => handleRepeat(order.id)}
              className="text-sm text-primary hover:text-primary-hover font-medium border border-primary/30 hover:border-primary px-4 py-2 rounded-lg transition-colors"
            >
              Repetir pedido
            </button>
          </div>
        )
      })}
    </div>
  )
}
```

```typescript
// app/api/admin/b2b-users/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { supabaseAdmin } from '@/lib/supabase/admin'
import { createServerSupabaseClient } from '@/lib/supabase/server'

async function requireAdmin(): Promise<NextResponse | null> {
  const supabase = await createServerSupabaseClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return NextResponse.json({ error: 'No autorizado' }, { status: 401 })
  }
  const { data: profile } = await supabase
    .from('user_profiles')
    .select('role, is_active')
    .eq('id', user.id)
    .single()
  if (!profile?.is_active || !['admin', 'manager'].includes(profile.role)) {
    return NextResponse.json({ error: 'Acceso denegado' }, { status: 403 })
  }
  return null
}

/**
 * POST /api/admin/b2b-users
 * Creates a B2B user: auth user + user_profile (b2b_client) + customer (business).
 * Body: { email, password, business_name, phone?, address?, city? }
 */
export async function POST(request: NextRequest) {
  const authCheck = await requireAdmin()
  if (authCheck) return authCheck

  try {
    const body = await request.json()
    const { email, password, business_name, phone, address, city } = body

    if (!email || !password || !business_name) {
      return NextResponse.json(
        { error: 'email, password y business_name son requeridos' },
        { status: 400 }
      )
    }

    // 1. Create auth user
    const { data: authData, error: authError } = await supabaseAdmin.auth.admin.createUser({
      email,
      password,
      email_confirm: true,
    })

    if (authError) {
      return NextResponse.json({ error: authError.message }, { status: 400 })
    }

    const userId = authData.user.id

    // 2. Create user_profile with b2b_client role
    const { error: profileError } = await supabaseAdmin
      .from('user_profiles')
      .insert({
        id: userId,
        role: 'b2b_client',
        first_name: business_name,
        last_name: null,
        phone: phone || null,
        is_active: true,
      })

    if (profileError) {
      await supabaseAdmin.auth.admin.deleteUser(userId)
      return NextResponse.json({ error: 'Error al crear perfil' }, { status: 500 })
    }

    // 3. Create customer record with type 'business'
    const { data: customer, error: customerError } = await supabaseAdmin
      .from('customers')
      .insert({
        user_id: userId,
        email,
        phone: phone || null,
        first_name: business_name,
        last_name: null,
        address: address || null,
        city: city || null,
        type: 'business',
        tags: ['b2b', 'cafeteria'],
      })
      .select()
      .single()

    if (customerError) {
      await supabaseAdmin.from('user_profiles').delete().eq('id', userId)
      await supabaseAdmin.auth.admin.deleteUser(userId)
      return NextResponse.json({ error: 'Error al crear cliente' }, { status: 500 })
    }

    return NextResponse.json({
      id: userId,
      email,
      business_name,
      customer_id: customer.id,
    })
  } catch (error: any) {
    console.error('Error in POST /api/admin/b2b-users:', error)
    return NextResponse.json({ error: 'Error interno' }, { status: 500 })
  }
}
```
